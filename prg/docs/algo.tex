\begin{algorithm}
  \caption{enum\_W}
  \begin{algorithmic}
    \REQUIRE $w,i,D,S$
    \STATE $w[i]\leftarrow (w[i]+1)\mod (i*S+1)$
    \IF{$w[i]=0$ \AND $i>1$}
    \STATE $w[i]\leftarrow$ enum\_W $w$ $(i-1)$ $d$ $s$
    \RETURN $w[i]$
    \ELSE
    \RETURN $w[i]$
    \ENDIF
  \end{algorithmic}
\end{algorithm}
